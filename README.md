- Exercise-1:
#### Swap the values of 2 variables
```js
  // takes two variables
let varOne = "Variable One";
let varTwo = 2;
//before swap
console.log("------------Before Swap ----------------");
console.log('Variable one =>', varOne);
console.log('Variable Two =>', varTwo);
//let's take a third variable for swap
let varSwap = varOne;
varOne = varTwo;
varTwo = varSwap;

// print the values
//after swap
console.log("------------After Swap ----------------");
console.log('Variable one =>', varOne);
console.log('Variable Two =>', varTwo);
```
